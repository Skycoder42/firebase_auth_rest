Future<RefreshResponse> token({
  @required String refresh_token,
  String grant_type = "refresh_token",
}) async =>
    RefreshResponse.fromJson(await _postQuery(
      _buildUri(
        "token",
        isTokenRequest: true,
      ),
      {
        "refresh_token": refresh_token,
        "grant_type": grant_type,
      },
    ));
